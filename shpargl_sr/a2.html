# запись в json файл, если дана строка, то преобразуем в словарь, если дан словарь, то dict_s = json.loads(s) не нужно
import json

s = '''{
    "name": "Barsik",
    "age": 7,
    "meals": [
        "Wiskas",
        "Royal Canin",
        "Purina",
        "Hills",
        "Brit Care"
    ]
}
'''

dict_s = json.loads(s)  # преобразования строки в словарь

with open('js.json', mode='w') as js_file:
    json.dump(dict_s, js_file, indent=2, ensure_ascii=True)
