import argparse

parser = argparse.ArgumentParser()
parser.add_argument("arg1")                           # собирает первую строку
parser.add_argument("int_args", type=int, nargs='*')  # собирает все числа
parser.add_argument("arg2")                           # собирает последнюю строку
args = parser.parse_args()

print(args.arg1, args.arg2, args.int_args)
