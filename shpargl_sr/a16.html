import requests

n = input()  # строка сервера
m = input()  # порт сервера
a = int(input())
b = int(input())

params = {  # параметры запроса
    "a": a,
    "b": b
}

response = requests.get(f'{n}:{m}', params=params).json()  # запрос с сервера в виде словаря
print(response)
