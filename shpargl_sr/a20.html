import csv
import json

a = []
with open('houses.csv', encoding="utf8") as csvfile:
    reader = csv.reader(csvfile, delimiter=';', quotechar='"')  # delimiter - разделитель, quotechar - кавычки
    f = True
    for index, row in enumerate(reader):
        if f:
            f = False
        else:
            if int(row[2]) != int(row[3]):
                a.append({
                    "adress": row[1],
                    "difference": abs(int(row[2]) - int(row[3]))
                })

with open('unusual.json', mode='w') as js_file:
    json.dump(a, js_file, indent=2, ensure_ascii=True)
